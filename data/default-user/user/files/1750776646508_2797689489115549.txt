  button.on('click', () => this.toggleUI());
  $('body').append(button);

  // Add keyboard shortcut (Ctrl+Shift+S or Cmd+Shift+S on Mac)
  $(document).on('keydown', (e) => {
    if ((e.ctrlKey || e.metaKey) && e.shiftKey && e.key === 'S') {
      e.preventDefault();
      this.toggleUI();
    }
  });

	/**
	 * Setup text filtering features
	 * @private
	 */
	_addItemTextSearchFilter() {
	  this._configurationManager.addTextField(
	    FILTER_TEXT_SEARCH,
	    'Search for videos containing specific text in their titles.'
	  );

	  this._addItemComplianceFilter(FILTER_TEXT_SEARCH, (item, searchText) => {
	    if (!searchText || searchText.trim() === '') {
	      return true;
	    }

	    const itemName = this._getItemName(item).toLowerCase();
	    const searchTerms = searchText.toLowerCase().split(' ');

	    return searchTerms.every(term => itemName.includes(term));
	  });
	}

	/**
	 * Setup text blacklist filtering
	 * @private
	 */
	_addItemBlacklistFilter(description) {
	  this._configurationManager.addTextAreaField(
	    FILTER_TEXT_BLACKLIST,
	    description
	  );

	  this._addItemComplianceFilter(FILTER_TEXT_BLACKLIST, (item, blacklistText) => {
	    if (!blacklistText || blacklistText.trim() === '') {
	      return true;
	    }

	    const itemName = this._getItemName(item).toLowerCase();
	    const blacklistTerms = blacklistText.toLowerCase().split('\n').filter(term => term.trim() !== '');

	    return !blacklistTerms.some(term => itemName.includes(term.trim()));
	  });
	}

	/**
	 * Setup text whitelist filtering
	 * @private
	 */
	_addItemWhitelistFilter(description) {
	  this._configurationManager.addTextAreaField(
	    FILTER_TEXT_WHITELIST,
	    description
	  );

	  this._addItemComplianceFilter(FILTER_TEXT_WHITELIST, (item, whitelistText) => {
	    if (!whitelistText || whitelistText.trim() === '') {
	      return true;
	    }

	    const itemName = this._getItemName(item).toLowerCase();
	    const whitelistTerms = whitelistText.toLowerCase().split('\n').filter(term => term.trim() !== '');

	    return whitelistTerms.some(term => itemName.includes(term.trim()));
	  });

	/**
	 * Setup text sanitization
	 * @private
	 */
	_addItemTextSanitizationFilter(description) {
	  this._configurationManager.addTextAreaField(
	    FILTER_TEXT_SANITIZATION,
	    description,
	    null,
	    (value) => {
	      let optimized = {};
	      if (value) {
	        let rules = value.split('\n');
	        for (let rule of rules) {
	          rule = rule.trim();
	          if (rule) {
	            let parts = rule.split('=');

	              let replacements = parts[1].split(',').map(r => r.trim());
	              optimized[target] = replacements;
	            }
	          }
	        }
	      }
	      return {
	        raw: value,
	        optimized: optimized
